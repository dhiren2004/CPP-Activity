/*5.	Consider a class Point containing x and y coordinates. Write a C++ program 
to implement necessary functions to accept a point, to display a point and to find 
distance between two points using operator overloading (-). (Use friend function) */


#include <iostream>
#include <cmath>
using namespace std;

class Point 
{
private:
    float x, y;

public:
    void accept() 
    {
        cout << "Enter x coordinate: ";
        cin >> x;
        cout << "Enter y coordinate: ";
        cin >> y;
    }

    void display() 
    {
        cout << "(" << x << ", " << y << ")";
    }

    friend float operator-(Point p1, Point p2);
};

float operator-(Point p1, Point p2) 
{
    float dx = p1.x - p2.x;
    float dy = p1.y - p2.y;
    return sqrt(dx * dx + dy * dy); 
}

int main() 
{
    Point p1, p2;
    float distance;

    cout << "Enter first point:\n";
    p1.accept();

    cout << "\nEnter second point:\n";
    p2.accept();

    cout << "\nFirst Point: ";
    p1.display();

    cout << "\nSecond Point: ";
    p2.display();
   
    distance = p1 - p2;

    cout << "\n\nDistance between two points = " << distance << endl;

    return 0;
}
